## Installation

The library is available on npm. You can install it using your preferred package manager.

<CodeBlockTabs noStyleSuffix>

<CodeBlockTabsList>
  <CodeBlockTabsTrigger value="npm">npm</CodeBlockTabsTrigger>
  <CodeBlockTabsTrigger value="pnpm">pnpm</CodeBlockTabsTrigger>
  <CodeBlockTabsTrigger value="yarn">yarn</CodeBlockTabsTrigger>
  <CodeBlockTabsTrigger value="bun">bun</CodeBlockTabsTrigger>
</CodeBlockTabsList>

<CodeBlockTab value="npm">

```bash
npm install @catcherjs/core
```

</CodeBlockTab>

<CodeBlockTab value="pnpm">

```bash
pnpm add @catcherjs/core
```

</CodeBlockTab>

<CodeBlockTab value="yarn">

```bash
yarn add @catcherjs/core
```

</CodeBlockTab>

<CodeBlockTab value="bun">

```bash
bun add @catcherjs/core
```

</CodeBlockTab>

</CodeBlockTabs>
